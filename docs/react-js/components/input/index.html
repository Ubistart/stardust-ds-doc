function example() {
  const [value, setValue] = React.useState("");
  const handleChange = (event) => setValue(event.target.value);


  return (
    <>
      <Typography mBottom="xxxs">Value: {value}</Typography>
      <Input
        isFullWidth
        value={value}
        onChange={handleChange}
        placeholder="Input controlado"
        iconRight={!!value.length ? <PhosphorIcons.X size={32} /> : null}
        iconRightAction={() => setValue("")}
      />
    </>
  );
}

Input com exemplo de senha​

Vamos usar esses componentes para criar um input com uma funcionalidade de mostrar/ocultar senha:

Resultado
Loading...
Editor em tempo real
function example(props) {
  const [show, setShow] = React.useState(false);
  const handleClick = () => setShow(!show);


  const canSeePasswordIcon = !show ? (
    <PhosphorIcons.Eye size={32} weight="light" color="white" />
  ) : (
    <PhosphorIcons.EyeSlash size={32} weight="light" />
  );


  return (
    <Input
      isFullWidth
      type={show ? "text" : "password"}
      placeholder="Input com exemplo de senha"
      iconRight={canSeePasswordIcon}
      iconRightAction={handleClick}
    />
  );
}

Input com icones​

Você pode adicionar ícones esquerdo e direito ao componente Input usando as propriedades iconLeft e iconRight respectivamente.

    placeholder="Input com ícone a esquerda"
    iconLeft={
      <PhosphorIcons.UserCircle size={32} weight="fill" color="black" />
    }
  />
  <Input
    isFullWidth
    placeholder="Input com ícone a direita"
    iconRight={
      <PhosphorIcons.UserCircle size={32} weight="fill" color="black" />
    }
  />
</div>
