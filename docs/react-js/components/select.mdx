## Importação

```tsx
import { Select } from "@stardust-ds/react";
```

## Uso

Exemplo de uso básico do componente de select:

```tsx live
<Select options={SELECT_OPTIONS.COLORS} placeholder="Selecione uma cor" />
```

### Select controlado

```tsx live
function example() {
  const [option, setOption] = React.useState({});

  return (
    <>
      <Typography mBottom="xxxs">Opção: {JSON.stringify(option)}</Typography>
      <Select
        options={SELECT_OPTIONS.COLORS}
        value={option}
        onSelect={(e) => setOption(e || {})}
        placeholder="Select controlado"
      />
    </>
  );
}
```

### Select com seções

É possível também criar seções de opções, ou seja, seção de cidades e cores.

```tsx live
<Select placeholder="Selecione uma opção" options={SELECT_OPTIONS.SECTIONS} />
```

### Select com multi seleção

```tsx live
<Select
  placeholder="Selecione uma ou mais opções"
  options={SELECT_OPTIONS.SECTIONS}
  multiSelect
/>
```

### Select com pesquisa

```tsx live
<Select
  placeholder="Select com pesquisa"
  options={SELECT_OPTIONS.SECTIONS}
  multiSelect
  searchable
/>
```

### Select sem opção de limpar o item selecionado

```tsx live
<Select
  placeholder="Select sem opção de limpeza"
  options={SELECT_OPTIONS.SECTIONS}
  multiSelect
  clearable={false}
/>
```

### Select com label e helper text

```tsx live
<Select
  label="Label"
  helperText="Helper Text"
  placeholder="Selecione uma opção"
  options={SELECT_OPTIONS.SECTIONS}
  noOptionsText="Nenhuma opção encontrada"
/>
```

### Select com erro

```tsx live
<Select
  label="Label"
  helperText="Helper Text"
  hasError
  placeholder="Selecione uma opção"
  options={SELECT_OPTIONS.SECTIONS}
  noOptionsText="Nenhuma opção encontrada"
/>
```

### Select sem nenhuma opção

```tsx live
<Select
  placeholder="Selecione uma opção"
  options={[]}
  noOptionsText="Nenhuma opção encontrada"
/>
```

### Interface de opções

```tsx
interface Option {
  label: string;
  value: string;
}
```

### Interface de opções com seções

```tsx
interface Section {
  section: string;
  options: Option[];
}
```
